## hashについて

hash = { name: "Taro" }

## hashの値追加方法

hash[追加するキー] = 値

teacher[:subject] = "English"

## hashの値取得方法

hash[取得したい値のキー]

puts teacher[:name]

## hashの値変更方法

hash[変更したい値のキー] = 値

teacher[:name] = "Emma"
